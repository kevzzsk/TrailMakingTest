function draw(ctx, location) {
    ctx.fillStyle = 'deepskyblue';
    ctx.shadowColor = 'dodgerblue';
    ctx.shadowBlur = 20;
    ctx.save();
    ctx.scale(SCALE, SCALE);
    ctx.translate(location.x / SCALE - OFFSET_X, location.y / SCALE - OFFSET_Y);
    ctx.fill(HOOK_PATH);
    ctx.restore();
}
